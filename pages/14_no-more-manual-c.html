#include <iostream>

using LL = long long;

constexpr int M = 1e9 + 7;

int main(){
    LL n; std::cin >> n;

    LL res = 0;

    for(LL f = 1; f <= n; ++f) res = (res + f * (n / f) % M) % M;

    std::cout << res << '\n';
}
